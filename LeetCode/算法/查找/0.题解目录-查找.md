public int researchBinary(int[] nums, target) {
  int right = nums.length - 1, left = 0;
  while(left <= right) {
    int mid = left + （right - left) / 2; //或(left + right) / 2或(left + right)>>1 
    if(nums[mid] == target)
      return mid;
    else if(nums[mid] < targrt)
      left = mid + 1;
    else
      right = mid - 1;
  }
  return left;
}
